        :: Create init: vue init webpack imagegal

        :: Install virtualenv: pip3 install virtualenv
        :: Create Virtual Env: virtualenv virt
        :: Run script: /virt/scripts/activate.bat
        :: Save Requirements: pip3 freeze > requirements.txt
